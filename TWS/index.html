Webhooks

Subscribe to any system event for external triggering.


// Example: Creating a custom workflow trigger
import { TWS } from '@tws/sdk';

const client = new TWS.Client({
  apiKey: process.env.TWS_API_KEY
});

client.events.on('invoice.created', async (invoice) => {
  if (invoice.total > 10000) {
    await client.workflows.trigger('cfo_approval', {
      invoiceId: invoice.id,
      priority: 'high'
    });
  }
});
